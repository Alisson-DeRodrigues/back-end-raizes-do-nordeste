npm init -y
npm install express jsonwebtoken bcryptjs pg cors
npm install -D typescript @types/express @types/jsonwebtoken @types/bcrypt @types/node @types/cors ts-node-dev

npm run dev

# Termux
pg_ctl -D $PREFIX/var/lib/postgresql start
pg_ctl -D $PREFIX/var/lib/postgresql stop
pg_ctl -D $PREFIX/var/lib/postgresql restart

pg_dump app > backup.sql
psql app < backup.sql

psql -U postgres -h localhost

createdb -U postgres meubanco
pg_dump -U postgres meubanco > backup.sql
psql -U postgres -d meubanco < backup.sql

sudo systemctl restart postgresql
sudo systemctl start postgresql
